PCA and the standard autoencoder are unsupervised dimensionality reduction methods, and their learned features are not discriminative. If you build a classifier upon the low-dimenional features extracted by PCA and autoencoder, you will find the classification accuracy very poor.

Linear discriminant analysis (LDA) is a traditionally supervised dimensionality reduction method for learning low-dimensional features which are highly discriminative. Likewise, can we extend autoencoder to supervised leanring?

You are required to build and train a supervised autoencoder look like the following. You are required to add other layers properly to alleviate overfitting.

import numpy

def to_one_hot(y, num_class=10):
    results = numpy.zeros((len(y), num_class))
    for i, label in enumerate(y):
        results[i, label] = 1.
    return results

from keras.layers import Dense, Input
from keras import models

input_img = Input(shape=(784,), name='input_img')

encode1 = Dense(512, activation='relu', name='encode1')(input_img)
encode2 = Dense(256, activation='relu', name='encode2')(encode1)
encode3 = Dense(128, activation='relu', name='encode3')(encode2)
encode4 = Dense(64, activation='relu', name='encode4')(encode3)
bottleneck = Dense(2, activation='relu', name='bottleneck')(encode4)
decode1 = Dense(64, activation='relu', name='decode1')(bottleneck)
decode2 = Dense(128, activation='relu', name='decode2')(decode1)
decode3 = Dense(256, activation='relu', name='decode3')(decode2)
decode4 = Dense(512, activation='relu', name='decode4')(decode3)

# build the supervised autoencoder network
from keras.layers import Dense, Input, Dropout, BatchNormalization, Activation
from keras import models

input_img = Input(shape=(784,), name='input_img')

# encoder network
encode1 = Dense(512, name='encode1')(input_img)
encode1 = BatchNormalization()(encode1)
encode1 = Activation('relu')(encode1)
encode2 = Dense(256, name='encode2')(encode1)
encode2 = BatchNormalization()(encode2)
encode2 = Activation('relu')(encode2)
encode3 = Dense(128, name='encode3')(encode2)
encode3 = BatchNormalization()(encode3)
encode3 = Activation('relu')(encode3)
encode4 = Dense(64, name='encode4')(encode3)
encode4 = BatchNormalization()(encode4)
encode4 = Activation('relu')(encode4)
# The width of the bottleneck layer must be exactly 2.
bottleneck = Dense(2, activation='relu', name='bottleneck')(encode4)

# decoder network
decode1 = Dense(64, name='decode1')(bottleneck)
decode1 = BatchNormalization()(decode1)
decode1 = Activation('relu')(decode1)
decode2 = Dense(128, name='decode2')(decode1)
decode2 = BatchNormalization()(decode2)
decode2 = Activation('relu')(decode2)
decode3 = Dense(256, name='decode3')(decode2)
decode3 = BatchNormalization()(decode3)
decode3 = Activation('relu')(decode3)
decode4 = Dense(512, name='decode4')(decode3)
decode4 = BatchNormalization()(decode4)
decode4 = Activation('relu')(decode4)
decode5 = Dense(784, activation='relu', name='decode5')(decode4)

# build a classifier upon the bottleneck layer

classifier1 = Dense(512, name='classifier1')(bottleneck)
classifier1 = BatchNormalization()(classifier1)
classifier1 = Activation('relu')(classifier1)

classifier2 = Dense(256, name='classifier2')(classifier1)
classifier2 = BatchNormalization()(classifier2)
classifier2 = Activation('relu')(classifier2)

classifier3 = Dense(128, name='classifier3')(classifier2)
classifier3 = BatchNormalization()(classifier3)
classifier3 = Activation('relu')(classifier3)

classifier4 = Dense(128, name='classifier4')(classifier3)
classifier4 = BatchNormalization()(classifier4)
classifier4 = Activation('relu')(classifier4)
classifier4 = Dropout(0.5)(classifier4)
classifier5 = Dense(10, activation='softmax', name='classifier5')(classifier4)
